cURL Request

curl https://your-api.onrender.com/v1/chat/completions \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer sk-aura-YOUR_KEY" \
  -d '{

Node.js (Fetch)

const response = await fetch("https://your-api.onrender.com/v1/chat/completions", {
  method: "POST",
  headers: {
    "Authorization": "Bearer sk-aura-YOUR_KEY",
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    model: "openrouter/free",
    messages: [{ role: "user", content: "Hello Aura!" }]
  })
});
console.log(await response.json());

Python (Requests)

import requests

url = "https://your-api.onrender.com/v1/chat/completions"
headers = {"Authorization": "Bearer sk-aura-YOUR_KEY", "Content-Type": "application/json"}
data = {"model": "openrouter/free", "messages": [{"role": "user", "content": "Hello!"}]}

res = requests.post(url, headers=headers, json=data)
print(res.json())

Java (OkHttp)

OkHttpClient client = new OkHttpClient();
MediaType JSON = MediaType.get("application/json; charset=utf-8");
String json = "{\"model\":\"openrouter/free\",\"messages\":[{\"role\":\"user\",\"content\":\"Hello\"}]}";

RequestBody body = RequestBody.create(json, JSON);
Request request = new Request.Builder()
  .url("https://your-api.onrender.com/v1/chat/completions")
  .post(body)
  .addHeader("Authorization", "Bearer sk-aura-YOUR_KEY")
  .build();

try (Response response = client.newCall(request).execute()) {
  System.out.println(response.body().string());
}
