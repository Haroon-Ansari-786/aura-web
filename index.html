Code Integration Examples

JavaScript (Fetch) 
const response = await fetch("https://your-api.onrender.com/v1/chat/completions", {
  method: "POST",
  headers: {
    "Authorization": "Bearer sk-aura-YOUR_API_KEY_HERE",
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    model: "openrouter/free",
    messages: [ { role: "user", content: "Hello Aura AI!" } ]
  })
});
console.log((await response.json()).choices[0].message.content);
Python 
import requests
import json

url = "https://your-api.onrender.com/v1/chat/completions"
headers = {
    "Authorization": "Bearer sk-aura-YOUR_API_KEY_HERE",
    "Content-Type": "application/json"
}
data = {
    "model": "openrouter/free",
    "messages": [{"role": "user", "content": "Hello!"}]
}
response = requests.post(url, headers=headers, json=data)
print(response.json()["choices"][0]["message"]["content"])
