Node.js - Chat Completion

const fetchAI = async () => {
  try {
    const response = await fetch("https://your-api.onrender.com/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": "Bearer sk-aura-YOUR_API_KEY",
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "openrouter/free",
        messages:[
          { role: "system", content: "You are Aura AI." },
          { role: "user", content: "Write a haiku about space." }
        ]
      })
    });
    
    const data = await response.json();
    console.log(data.choices[0].message.content);
  } catch (err) {
    console.error(err);
  }
};

fetchAI();

Python (Requests) - Image Generation

import requests

url = "https://your-api.onrender.com/v1/images/generations"
headers = {
    "Authorization": "Bearer sk-aura-YOUR_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "prompt": "A futuristic cyberpunk city, 8k resolution, highly detailed"
}

response = requests.post(url, headers=headers, json=payload)

if response.status_code == 200:
    data = response.json()
    print("Image URL:", data["data"][0]["url"])
else:
    print("Error:", response.text)

PHP (cURL) - Video Generation

<?php
$curl = curl_init();

curl_setopt_array($curl, array(
  CURLOPT_URL => 'https://your-api.onrender.com/v1/video/generations',
  CURLOPT_RETURNTRANSFER => true,
  CURLOPT_ENCODING => '',
  CURLOPT_MAXREDIRS => 10,
  CURLOPT_TIMEOUT => 0,
  CURLOPT_FOLLOWLOCATION => true,
  CURLOPT_HTTP_VERSION => CURL_HTTP_VERSION_1_1,
  CURLOPT_CUSTOMREQUEST => 'POST',
  CURLOPT_POSTFIELDS =>'{"prompt":"A dog running in the park cinematic"}',
  CURLOPT_HTTPHEADER => array(
    'Authorization: Bearer sk-aura-YOUR_API_KEY',
    'Content-Type: application/json'
  ),
));

$response = curl_exec($curl);
curl_close($curl);
echo $response;
?>

Java (OkHttp) - General Request

OkHttpClient client = new OkHttpClient();
MediaType JSON = MediaType.get("application/json; charset=utf-8");
String json = "{\"model\":\"openrouter/free\",\"messages\":[{\"role\":\"user\",\"content\":\"Hello\"}]}";

RequestBody body = RequestBody.create(json, JSON);
Request request = new Request.Builder()
  .url("https://your-api.onrender.com/v1/chat/completions")
  .post(body)
  .addHeader("Authorization", "Bearer sk-aura-YOUR_KEY")
  .build();

try (Response response = client.newCall(request).execute()) {
  System.out.println(response.body().string());
}
